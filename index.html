    // ─── STATE ───────────────────────────────────────────────────────────────────
    let state = {
    screen: 'start', // start, playing, results
    difficulty: 2, // 1=leicht, 2=mittel, 3=schwer
    questions: [],
    current: 0,
    selected: null,
    score: 0,
    streak: 0,
    maxStreak: 0,
    showExplanation: false,
    animating: false,
    };
    function getQuestionCount() {
    return state.difficulty === 1 ? 8 : state.difficulty === 2 ? 10 : 12;
    }
    function startGame() {
    const count = getQuestionCount();
    const maxDiff = state.difficulty;
    const filtered = QUESTIONS.filter(q => q.difficulty <= maxDiff);
    const shuffled = filtered.sort(() => Math.random() - 0.5).slice(0, count);
    state = {
    ...state, screen: 'playing', questions: shuffled,
    current: 0, selected: null, score: 0, streak: 0, maxStreak: 0,
    showExplanation: false, animating: false,
    };
    render();
    }
    function selectOption(idx) {
    if (state.selected !== null || state.animating) return;
    state.selected = idx;
    state.showExplanation = true;
    const q = state.questions[state.current];
    if (idx === q.correct) {
    state.score++;
    state.streak++;
    state.maxStreak = Math.max(state.maxStreak, state.streak);
    } else {
    state.streak = 0;
    }
    render();
    }
    function nextQuestion() {
    if (state.current + 1 >= state.questions.length) {
    state.screen = 'results';
    } else {
    state.animating = true;
    state.current++;
    state.selected = null;
    state.showExplanation = false;
    state.animating = false;
    }
    render();
    }
    function setDifficulty(d) {
    state.difficulty = d;
    render();
    }
    function restart() {
    state.screen = 'start';
    render();
    }
    // ─── RENDER ──────────────────────────────────────────────────────────────────

    const cat = CATEGORIES.find(c => c.name === name);
    return cat ? `background:${cat.bg}; color:${cat.color}; border: 1px solid ${cat.color}30;` : '';
    }
    function getCatColor(name) {
    const cat = CATEGORIES.find(c => c.name === name);
    return cat ? cat.color : '#888';
    }

    const app = document.getElementById('app');
    if (state.screen === 'start') {
    app.innerHTML = `
